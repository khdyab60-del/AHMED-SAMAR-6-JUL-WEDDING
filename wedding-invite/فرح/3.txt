<script>
const audio = document.getElementById("bg-music");
const iconOn = document.getElementById("icon-on");
const iconOff = document.getElementById("icon-off");

let started = false;

function startMusic() {
  if (started) return;
  
  // رفع الحظر عن الصوت عبر التفاعل الأول للمستخدم
  audio.play()
    .then(() => {
      started = true;
      iconOn.style.display = "block";
      iconOff.style.display = "none";
      // إزالة المستمعين فور نجاح التشغيل لتوفير الأداء
      document.removeEventListener("click", startMusic);
      document.removeEventListener("touchstart", startMusic);
    })
    .catch((error) => {
      console.log("المتصفح يمنع التشغيل التلقائي حتى الآن:", error);
    });
}

// الاستماع لأي نقرة أو لمسة في أي مكان بالصفحة لتشغيل الصوت فوراً
document.addEventListener("click", startMusic, { once: true });
document.addEventListener("touchstart", startMusic, { once: true });

function toggleSound() {
  // إذا لم يكن قد بدأ بعد، نقوم بتشغيله
  if (!started) {
    audio.play().then(() => { started = true; });
  }

  if (audio.paused) {
    audio.play();
    iconOn.style.display = "block";
    iconOff.style.display = "none";
  } else {
    audio.pause();
    iconOn.style.display = "none";
    iconOff.style.display = "block";
  }
}

    var weddingDate = new Date("July 6, 2026 20:00:00").getTime();

    var x = setInterval(function () {

      var distance = weddingDate - now;

      if (distance < 0) {
        clearInterval(x);
        document.getElementById("countdown").innerHTML = "Just Married! 💍";
        return;
      }

      var days = Math.floor(distance / (1000 * 60 * 60 * 24));
      var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
      var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
      var seconds = Math.floor((distance % (1000 * 60)) / 1000);

      document.getElementById("countdown").innerHTML =
        days + "D " + hours + "H " + minutes + "M " + seconds + "S";
    }, 1000);
  </script>

function addGuest() {
  // التحديث مباشرة بخطوة واحدة لمنع تداخل البيانات وضياع التبريكات
  fetch(apiEndpoint, {
    method: 'PUT',
    headers: {
      'Content-Type': 'application/json',
      'X-Master-Key': apiKey,
      'X-Bin-Versioning': 'false' // لتحديث نفس النسخة دون إنشاء نسخ احتياطية زائدة
    },
    body: JSON.stringify({
      // السيرفر سيقوم بزيادة الرقم مباشرة بناءً على القيمة الحالية الموثوقة لديه
      count: parseInt(document.getElementById("guestCount").innerText) + 1
    })
  })
  .then(res => {
    if (!res.ok) throw new Error();
    return res.json();
  })
  .then(data => {
    // تحديث الشاشة بالقيمة الجديدة المستقرة من السيرفر
    document.getElementById("guestCount").innerText = data.record.count;
    localStorage.setItem("guestCount", data.record.count);
  })
  .catch(() => {
    // حل احتياطي سريع في حال انقطاع اتصال الإنترنت لدى الضيف
    let count = parseInt(localStorage.getItem("guestCount") || 0) + 1;
    localStorage.setItem("guestCount", count);
    document.getElementById("guestCount").innerText = count;
  });
}

function triggerFireworks() {
  var duration = 2.5 * 1000;
  var end = Date.now() + duration;

  (function frame() {
    confetti({ particleCount: 4, angle: 60, spread: 55, origin: { x: 0, y: 0.85 } });
    confetti({ particleCount: 4, angle: 120, spread: 55, origin: { x: 1, y: 0.85 } });
    if (Date.now() < end) {
      requestAnimationFrame(frame);
    }
  }());
}

function handleCongrats() {
  // تشغيل المؤثرات البصرية دائماً عند الضغط
  triggerFireworks();

  // نظام حماية يمنع نفس الشخص من الضغط وإفساد الإحصائيات الفعليّة لأكثر من مرة
  if (localStorage.getItem("hasCongratulated") === "true") {
    console.log("تم تسجيل تبريكاتك مسبقاً، شكراً لك!");
    return; 
  }

  addGuest();
  localStorage.setItem("hasCongratulated", "true");
}
